class SocketProxy:
    def __init__(self, sock, alloc_fd, method_name, stack_summary):
        self._sock = sock
        self._alloc_fd = alloc_fd
        self._method = method_name
        self._stack_summary = stack_summary
        # This triggers when the object is actually destroyed by GC
        weakref.finalize(self, self._log_death, alloc_fd)

    @staticmethod
    def _log_death(fd):
        print(f"[DELETE] Socket object for FD {fd} removed from memory.")

    def close(self):
        fd = self._sock.fileno()
        result = self._sock.close()
        print(f"[CLOSE] FD: {fd if fd != -1 else self._alloc_fd}")
        return result

    def __getattr__(self, name):
        return getattr(self._sock, name)

class SocketTracker:
    _originals = {'socket': socket.socket, 'fromfd': socket.fromfd}

    @classmethod
    def install(cls):
        socket.socket = cls._tracked_socket
        socket.fromfd = cls._tracked_fromfd
        print("\n[SocketTracker] Full Stack Monitoring started...")

    @classmethod
    def uninstall(cls):
        socket.socket = cls._originals['socket']
        socket.fromfd = cls._originals['fromfd']
        print("\n[SocketTracker] Monitoring stopped.")

    @staticmethod
    def _log_alloc(method_name, real_sock):
        fd = real_sock.fileno()
        stack = traceback.extract_stack()
        
        # Filter: Keep your test, cheroot, and urllib3; ignore tracker/std-lib internal wrappers
        relevant_frames = [
            f for f in stack 
            if ("test_server.py" in f.filename or "cheroot" in f.filename or "urllib3" in f.filename)
            and "SocketTracker" not in f.name
            and "_log_alloc" not in f.name
        ]

        print(f"\n[ALLOC] {method_name.upper()} | FD: {fd}")
        print("   Traceback (User -> Lib):")

        show_full = False
        indent = ""
        for i, frame in enumerate(relevant_frames):
            if show_full:
                indent = "  " * (i + 1)
            
            # Mark your test file with an arrow
            marker = "→ " if "test_server.py" in frame.filename else "  "
            if show_full or "test_server.py" in frame.filename:
                print(f"{indent}{marker}{os.path.basename(frame.filename)}:{frame.lineno} in {frame.name}")
        
        return SocketProxy(real_sock, fd, method_name, relevant_frames)

    @staticmethod
    def _tracked_socket(*args, **kwargs):
        return SocketTracker._log_alloc('socket', SocketTracker._originals['socket'](*args, **kwargs))

    @staticmethod
    def _tracked_fromfd(*args, **kwargs):
        return SocketTracker._log_alloc('fromfd', SocketTracker._originals['fromfd'](*args, **kwargs))

def get_open_fds():
    """Count actual open file descriptors for the current process."""
    try:
        # Standard way on Linux to see open files for the current process
        return len(list(Path('/proc/self/fd').iterdir()))
    except (FileNotFoundError, PermissionError):
        # Fallback for macOS or environments without /proc
        return 0


@pytest.hookimpl(hookwrapper=True)
def pytest_runtest_protocol(item, nextitem):
    """Check for leaks."""
    print(f'\n[LEAK CHECK] Starting test: {item.name}')
    if hasattr(item.config, 'workerinput'):  # running under xdist, skip
        yield
        return

    # Capture state BEFORE the test
    start_fds = get_open_fds()

    yield

    # Capture state AFTER the test and teardown
    gc.collect()  # Force GC to trigger ResourceWarnings NOW
    end_fds = get_open_fds()

    if end_fds > start_fds:
        print(
            f'\n[FD LEAK] {item.name}: Started with {start_fds}, ended with {end_fds}',
        )

def log_leaking_socket_refs():
    """Scan for unclosed sockets and log their referrers to CI stdout."""
    import inspect  # Needed for frame-checking in the helper
    import sys
    import warnings

    divider = '=' * DIVIDER_SIDE_WIDTH
    # noqa: WPS421
    print(f'\n{divider} CI RESOURCE LEAK DEBUGGER {divider}')

    # Silence the warning during the initial collection
    with warnings.catch_warnings():
        warnings.simplefilter('ignore', ResourceWarning)
        gc.collect()

    for gc_obj in gc.get_objects():
        if not isinstance(gc_obj, socket.socket):
            continue

        with suppress(OSError, AttributeError):
            if gc_obj.fileno() == -1:
                continue

            print(f'\n[LEAK DETECTED] Found unclosed socket: {gc_obj}')
            # Pass the inspect module and start at max depth
            _log_referrers(gc_obj, inspect, depth=2)

    divider_footer = '=' * DIVIDER_TOTAL_WIDTH
    print(f'{divider_footer}\n')
    sys.stdout.flush()


def _log_referrers(gc_obj, inspect_mod, depth=2):  # noqa: WPS231, WPS210
    """Recursively find who owns the containers holding the socket."""
    referrers = gc.get_referrers(gc_obj)

    for ref in referrers:
        # Skip internal GC structures and frames to keep logs clean
        if ref is referrers or inspect_mod.isframe(ref):
            continue

        ref_type = type(ref)
        indent_count = 2 - depth
        padding = '  ' * indent_count
        print(f'{padding}-> Held by: {ref_type}')

        if isinstance(ref, dict):
            # If it's a __dict__, find the object it belongs to
            owner = _find_dict_owner(ref)
            if owner:
                print(f'     Owner Object: {type(owner)} - {owner}')  # noqa: WPS421
            else:
                keys_sample = list(ref.keys())[:5]
                print(f'     Keys: {keys_sample}')  # noqa: WPS421

        # Climb one level higher if we are still looking at a container
        if depth > 0 and isinstance(ref, (list, dict, tuple)):
            _log_referrers(ref, inspect_mod, depth - 1)


def _find_dict_owner(target_dict):
    """Find the object instance that owns this __dict__."""
    for gc_obj in gc.get_objects():
        with suppress(AttributeError, EnvironmentError):
            if getattr(gc_obj, '__dict__', None) is target_dict:
                return gc_obj
    return None

class TraceableSocket(_original_socket):
    """A socket subclass that records its creation traceback for leak detection."""

    def __init__(self, *args, **kwargs):
        """Initialize the TraceableSocket and record its creation traceback."""
        super().__init__(*args, **kwargs)
        self._birth_certificate = ''.join(traceback.format_stack())

    def accept(self):
        """Override accept to track accepted sockets."""
        fd, addr = super().accept()
        fd._birth_certificate = ''.join(traceback.format_stack())
        return fd, addr
